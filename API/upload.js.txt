import { Dropbox } from "dropbox";
import formidable from "formidable";
import fs from "fs";

export const config = {
  api: {
    bodyParser: false,
  },
};

const DROPBOX_ACCESS_TOKEN = process.env.DROPBOX_ACCESS_TOKEN;

export default async function handler(req, res) {
  if (req.method !== "POST") {
    res.status(405).send("Method Not Allowed");
    return;
  }

  const dbx = new Dropbox({ accessToken: DROPBOX_ACCESS_TOKEN, fetch: fetch });

  const form = new formidable.IncomingForm();
  form.parse(req, async (err, fields, files) => {
    if (err) {
      res.status(500).json({ error: "Error parsing the files" });
      return;
    }

    const file = files.photo;
    if (!file) {
      res.status(400).json({ error: "No file uploaded" });
      return;
    }

    try {
      const data = await fs.promises.readFile(file.filepath);
      const filename = `reni_${Date.now()}_${file.originalFilename}`;

      await dbx.filesUpload({
        path: `/FotosCumple/${filename}`,
        contents: data,
      });

      res.status(200).json({ message: "File uploaded successfully" });
    } catch (error) {
      console.error(error);
      res.status(500).json({ error: "Error uploading file to Dropbox" });
    }
  });
}
